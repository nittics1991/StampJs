		line_weight:3,

		text_direction:'V',
		option_position:'bottom',
        backgroound_color:'white',
	},

		{
			stamp_text:'太一郎',
			font_family:'',
			font_size:30,
			text_align:'center',
			vertical_align:'middle',
			margin_size:0,
		},
		{
			stamp_text:'青木',
			font_family:'',
			font_size:38,
			text_align:'center',
			vertical_align:'middle',
			margin_size:0,
		},
		{
			stamp_text:'代',
			font_family:null,
			font_size:30,
			text_align:'right',
			vertical_align:'bottom',
			margin_size:0,
		},
	],
};


var obj = new StampJs('#date_stamp', dataset);
obj.render();
